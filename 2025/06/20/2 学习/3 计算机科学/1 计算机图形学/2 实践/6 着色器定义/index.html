创建简单着色器

在 cmake-build-debug 路径下创建 assets 文件夹

新建着色器文件 simple.frag 和 simple.vert，意思是简单的着色器，没有光照计算，纹理计算等。

frag 和 vert 文件

frag 是片元着色器，vert 是顶点着色器

.vert 文件

设置：

  • 模型矩阵 model（来自 world）
  • 视图矩阵 view（来自 camera）
  • 投影矩阵 projection（来自 camera）

输入：

  • 点坐标、颜色

输出：

  • 点的剪裁空间位置
  • 点颜色
// 指定 GLSL 版本为 3.30，并使用核心模式(Core-profile)
#version 330 core

// --- 输入变量 (Vertex Attributes) ---
// 从 VBO 接收顶点数据。layout(location = N) 将变量绑定到第 N 个顶点属性。

// layout(location = 0): 绑定到第0个属性，通常是位置数据
// in vec3: 输入一个三维向量
// aPos: 变量名，代表顶点位置 (Position)
layout (location = 0) in vec3 aPos;

// layout(location = 1): 绑定到第1个属性，通常是颜色数据
// aColor: 变量名，代表顶点颜色 (Color)
layout (location = 1) in vec3 aColor;


// --- 输出变量 ---
// 将数据传递给下一个着色器阶段（片元着色器）

// out vec3: 输出一个三维向量
// ourColor: 变量名，用于将颜色传递出去
out vec3 ourColor;


// --- 全局变量 (Uniforms) ---
// 从 CPU 程序传入的、对所有顶点都相同的全局变量

// mat4: 4x4 矩阵类型
// model: 模型矩阵 (将模型局部坐标 -> 世界坐标)
// view: 视图矩阵 (将世界坐标 -> 观察空间坐标)
// projection: 投影矩阵 (将观察空间坐标 -> 裁剪空间坐标)
uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;


// --- 主函数 ---
// 每个顶点都会执行一次 main 函数
void main()
{
    // gl_Position 是一个内置的、必须被赋值的 vec4 输出变量
    // 它决定了顶点在屏幕上的最终位置
    // 计算过程：通过 MVP 矩阵链式相乘，将顶点从局部空间变换到裁剪空间
    // 注意：矩阵乘法顺序为 投影 * 视图 * 模型
    // vec4(aPos, 1.0) 是将三维位置坐标转换为四维齐次坐标，w分量为1.0
    gl_Position = projection * view * model * vec4(aPos, 1.0);

    // 将输入的顶点颜色直接赋值给输出变量
    // 这个颜色值会经过插值后，传递给片元着色器
    ourColor = aColor;
}

// 总结：
// 1. 主要任务：计算每个顶点的最终裁剪空间位置 (gl_Position)。
// 2. 附加任务：将顶点颜色 (aColor) 原封不动地传递给片元着色器 (ourColor)。

.frag 文件

片元上色

参数：

  • RGB 值

输出：

  • 片元颜色 RGBA
#version 330 core
out vec4 FragColor;

in vec3 ourColor;

void main()
{
    // 将输入的RGB颜色和1.0的Alpha值组合成一个vec4，作为最终颜色输出
    FragColor = vec4(ourColor, 1.0f);
}

着色器定义

着色器类用于确定一种着色方法。比如在上面使用最简单的着色方法，包含了片元着色器和顶点着色器。

所以对于一个着色方法，是多个着色器的组合。创建一个类将他们整合起来。

着色器定义如下

//=============== 文件路径: components/shader/Shader.h ===============//
#ifndef SHADER_H
#define SHADER_H

#include "GL/glew.h"
#include "glm/glm.hpp"

#include <string>
#include <fstream>
#include <sstream>
#include <iostream>

class Shader
{
public:
    // 着色器程序ID
    unsigned int ID;

    // 构造函数：从文件读取并构建着色器
    Shader(const char* vertexPath, const char* fragmentPath);

    // 激活着色器程序
    void use();

    // uniform工具函数
    void setBool(const std::string &name, bool value) const;
    void setInt(const std::string &name, int value) const;
    void setFloat(const std::string &name, float value) const;
    void setMat4(const std::string &name, const glm::mat4 &mat) const;

private:
    // 检查编译/链接错误的工具函数
    void checkCompileErrors(unsigned int shader, std::string type);
};

#endif

实现如下

//=============== 文件路径: components/shader/Shader.cpp ===============//
#include "Shader.h"

Shader::Shader(const char* vertexPath, const char* fragmentPath)
{
    // 1. 从文件路径中获取顶点/片元着色器源代码
    std::string vertexCode;
    std::string fragmentCode;
    std::ifstream vShaderFile;
    std::ifstream fShaderFile;

    // 保证ifstream对象可以抛出异常
    vShaderFile.exceptions(std::ifstream::failbit | std::ifstream::badbit);
    fShaderFile.exceptions(std::ifstream::failbit | std::ifstream::badbit);
    try
    {
        // 打开文件
        vShaderFile.open(vertexPath);
        fShaderFile.open(fragmentPath);
        std::stringstream vShaderStream, fShaderStream;
        // 读取文件缓冲内容到数据流中
        vShaderStream << vShaderFile.rdbuf();
        fShaderStream << fShaderFile.rdbuf();
        // 关闭文件处理器
        vShaderFile.close();
        fShaderFile.close();
        // 转换数据流到string
        vertexCode = vShaderStream.str();
        fragmentCode = fShaderStream.str();
    }
    catch (std::ifstream::failure& e)
    {
        std::cerr << "ERROR::SHADER::FILE_NOT_SUCCESSFULLY_READ: " << e.what() << std::endl;
    }
    const char* vShaderCode = vertexCode.c_str();
    const char* fShaderCode = fragmentCode.c_str();

    // 2. 编译着色器
    unsigned int vertex, fragment;
    // 顶点着色器
    vertex = glCreateShader(GL_VERTEX_SHADER);
    glShaderSource(vertex, 1, &vShaderCode, NULL);
    glCompileShader(vertex);
    checkCompileErrors(vertex, "VERTEX");
    // 片元着色器
    fragment = glCreateShader(GL_FRAGMENT_SHADER);
    glShaderSource(fragment, 1, &fShaderCode, NULL);
    glCompileShader(fragment);
    checkCompileErrors(fragment, "FRAGMENT");

    // 着色器程序
    ID = glCreateProgram();
    glAttachShader(ID, vertex);
    glAttachShader(ID, fragment);
    glLinkProgram(ID);
    checkCompileErrors(ID, "PROGRAM");

    // 删除着色器，它们已经链接到我们的程序中了，不再需要
    glDeleteShader(vertex);
    glDeleteShader(fragment);
}

void Shader::use()
{
    glUseProgram(ID);
}

void Shader::setBool(const std::string &name, bool value) const
{
    glUniform1i(glGetUniformLocation(ID, name.c_str()), (int)value);
}

void Shader::setInt(const std::string &name, int value) const
{
    glUniform1i(glGetUniformLocation(ID, name.c_str()), value);
}

void Shader::setFloat(const std::string &name, float value) const
{
    glUniform1f(glGetUniformLocation(ID, name.c_str()), value);
}

void Shader::setMat4(const std::string &name, const glm::mat4 &mat) const
{
    glUniformMatrix4fv(glGetUniformLocation(ID, name.c_str()), 1, GL_FALSE, &mat[0][0]);
}

void Shader::checkCompileErrors(unsigned int shader, std::string type)
{
    int success;
    char infoLog[1024];
    if (type != "PROGRAM")
    {
        glGetShaderiv(shader, GL_COMPILE_STATUS, &success);
        if (!success)
        {
            glGetShaderInfoLog(shader, 1024, NULL, infoLog);
            std::cerr << "ERROR::SHADER_COMPILATION_ERROR of type: " << type << "\n" << infoLog << "\n -- --------------------------------------------------- -- " << std::endl;
        }
    }
    else
    {
        glGetProgramiv(shader, GL_LINK_STATUS, &success);
        if (!success)
        {
            glGetProgramInfoLog(shader, 1024, NULL, infoLog);
            std::cerr << "ERROR::PROGRAM_LINKING_ERROR of type: " << type << "\n" << infoLog << "\n -- --------------------------------------------------- -- " << std::endl;
        }
    }
}

着色器管理

创建一个映射，将一种着色方法对应到 Shader 上即可

#pragma once

#include <string>
#include <map>
#include <memory>
#include "Shader.h"

class ShaderManager {
public:
    // 按需获取或加载着色器
    Shader* getOrLoadShader(const std::string& name);

private:
    std::map<std::string, std::unique_ptr<Shader>> m_Shaders;
};

实时创建加载着色器。在需要用到时再加载着色器。

#include "ShaderManager.h"
#include "Shader.h"
#include "../AssetsPath.h"
#include <iostream>

// 实现 getOrLoadShader
Shader* ShaderManager::getOrLoadShader(const std::string& name) {
    // 1. 检查缓存中是否已存在
    if (m_Shaders.find(name) != m_Shaders.end()) {
        return m_Shaders[name].get();
    }

    // 2. 如果不存在，则根据约定构建文件路径
    std::cout << "Info: [ShaderManager] Cache miss for '" << name << "'. Loading from conventional path." << std::endl;
    std::string vertexPath = AssetPaths::SHADER_DIR + name + ".vert";
    std::string fragmentPath = AssetPaths::SHADER_DIR + name + ".frag";

    // 3. 尝试加载和创建Shader
    //    这里我们把创建逻辑包在 try-catch 块中，因为Shader构造函数可能会因文件不存在而抛出异常
    try {
        auto newShader = std::make_unique<Shader>(vertexPath.c_str(), fragmentPath.c_str());
        // 如果 newShader 的 ID 是0，可能表示链接失败但未抛出异常（取决于Shader类的实现）
        // 最好在Shader构造函数中如果失败就抛异常或返回nullptr
        if (newShader && newShader->ID != 0) {
            m_Shaders[name] = std::move(newShader);
            std::cout << "Info: [ShaderManager] Shader loaded and cached: " << name << std::endl;
            return m_Shaders[name].get();
        }
    } catch (const std::ifstream::failure& e) {
        // Shader构造函数中的文件读取异常会被这里捕获
        std::cerr << "Error: [ShaderManager] Failed to load shader files for '" << name << "'. Exception: " << e.what() << std::endl;
        return nullptr;
    }

    // 如果因其他原因（如编译或链接失败）创建失败
    std::cerr << "Error: [ShaderManager] Failed to create shader program for '" << name << "'." << std::endl;
    return nullptr;
}
