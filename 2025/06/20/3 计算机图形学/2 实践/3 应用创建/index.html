定义 Application

//=============== 文件路径: Application.h ===============//
class Application {
public:
    Application(unsigned int width, unsigned int height);
    ~Application();
    void Run();

private:
    void init();
    void processInput();
    void update(float deltaTime);
    void render();

    // 鼠标回调的成员函数
    void handleMouse(float xpos, float ypos);

    // 转发到成员函数的静态回调
    static void framebuffer_size_callback(GLFWwindow* window, int width, int height);
    static void mouse_callback(GLFWwindow* window, double xpos, double ypos);

    unsigned int m_Width, m_Height;
    GLFWwindow* m_Window;

    ShaderManager m_ShaderManager;
    MeshManager m_MeshManager;
    World m_World;

    Camera m_Camera;
    float m_LastX;
    float m_LastY;
    bool m_FirstMouse = true;

    float m_DeltaTime = 0.0f;
    float m_LastFrame = 0.0f;
};

构造参数：

  • 窗口宽和高
  • 相机
  • 上一次鼠标的位置
//=============== 文件路径: Application.cpp ===============//
#include "Application.h"
#include <iostream>
#include <glm/gtc/matrix_transform.hpp>
#include "objects/Object.h"
#include "scene/SceneSerializer.h"

Application::Application(unsigned int width, unsigned int height)
        : m_Width(width),
          m_Height(height),
          m_Window(nullptr),
          m_Camera(glm::vec3(0.0f, 1.0f, 6.0f)), // 将相机向后移动一点以便观察
          m_LastX(width / 2.0f),
          m_LastY(height / 2.0f) {
    init();
}

Application::~Application() {
    // World的析构函数会自动清理所有GameObject
    glfwTerminate();
}

void Application::init() {
    // --- GLFW, Window, GLEW 初始化---
    if (!glfwInit()) {
        std::cerr << "Failed to initialize GLFW" << std::endl;
        return;
    }
    glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 3);
    glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 3);
    glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);
#ifdef __APPLE__
    glfwWindowHint(GLFW_OPENGL_FORWARD_COMPAT, GL_TRUE);
#endif

    m_Window = glfwCreateWindow(m_Width, m_Height, "fuckcraft", NULL, NULL);
    if (m_Window == NULL) {
        std::cerr << "Failed to create GLFW window" << std::endl;
        glfwTerminate();
        return;
    }
    glfwMakeContextCurrent(m_Window);
    glfwSetWindowUserPointer(m_Window, this);
    glfwSetFramebufferSizeCallback(m_Window, framebuffer_size_callback);
    glfwSetCursorPosCallback(m_Window, mouse_callback);
    glfwSetInputMode(m_Window, GLFW_CURSOR, GLFW_CURSOR_DISABLED);

    if (glewInit() != GLEW_OK) {
        std::cerr << "Failed to initialize GLEW" << std::endl;
        return;
    }

    glEnable(GL_DEPTH_TEST);

    // --- 加载场景 ---
    auto& factory = ObjectFactory::GetInstance();
    SceneSerializer serializer(m_World, factory, m_ShaderManager, m_MeshManager);
    if (!serializer.deserialize("assets/scene.json")) {
        std::cerr << "Error: Application could not initialize scene." << std::endl;
    }
}

执行循环

计算每帧耗时

将耗时传入更新逻辑，把每帧耗时作为系数乘上，数值的累加将会和帧率无关。也就是帧率高和低的电脑数值变化是一样的。

例如

void update(float deltaTime) {
    float speed = 100.0f; // 定义速度为每秒100像素
    character.x += speed * deltaTime; // 每帧的移动距离 = 速度 × 每帧耗时
}

循环应该按照如下顺序：

  • 先处理外部影响
  • 再更新内部状态
  • 最后展示结果
void Application::Run() {
    //检查用户是否点击关闭窗口
    while (!glfwWindowShouldClose(m_Window)) {
        //计算每帧耗时准备传给 update
        float currentFrame = static_cast<float>(glfwGetTime());
        m_DeltaTime = currentFrame - m_LastFrame;
        m_LastFrame = currentFrame;

        //输入处理
        processInput();
        //逻辑更新
        update(m_DeltaTime);
        //渲染
        render();

        //交换缓冲区（双缓冲技术）
        glfwSwapBuffers(m_Window);
        //轮询事件：点击关闭、键盘按下、鼠标移动、窗口大小改变
        glfwPollEvents();
    }
}

输入处理

键盘使用轮询的方式，而鼠标使用（软件层面的）回调的方式

键盘

按下 WASD 按键，移动的距离是固定的，所以只需要检查按键是否按下，每一帧轮询按照固定速度移动摄像机。

void Application::processInput() {
    if (glfwGetKey(m_Window, GLFW_KEY_ESCAPE) == GLFW_PRESS)
        glfwSetWindowShouldClose(m_Window, true);

    if (glfwGetKey(m_Window, GLFW_KEY_W) == GLFW_PRESS)
        m_Camera.ProcessKeyboard(FORWARD, m_DeltaTime);
    if (glfwGetKey(m_Window, GLFW_KEY_S) == GLFW_PRESS)
        m_Camera.ProcessKeyboard(BACKWARD, m_DeltaTime);
    if (glfwGetKey(m_Window, GLFW_KEY_A) == GLFW_PRESS)
        m_Camera.ProcessKeyboard(LEFT, m_DeltaTime);
    if (glfwGetKey(m_Window, GLFW_KEY_D) == GLFW_PRESS)
        m_Camera.ProcessKeyboard(RIGHT, m_DeltaTime);
    
    if (glfwGetKey(m_Window, GLFW_KEY_SPACE) == GLFW_PRESS)
        m_Camera.ProcessKeyboard(UP, m_DeltaTime);
    if (glfwGetKey(m_Window, GLFW_KEY_LEFT_SHIFT) == GLFW_PRESS)
        m_Camera.ProcessKeyboard(DOWN, m_DeltaTime);
}

鼠标

移动鼠标，移动的幅度会决定视角移动距离，此时会附带上下文信息，使用回调函数注册方式。

此处计算鼠标移动距离传递给摄像机。

void Application::mouse_callback(GLFWwindow *window, double xpos, double ypos) {
    Application *app = static_cast<Application *>(glfwGetWindowUserPointer(window));
    if (app) {
        app->handleMouse(static_cast<float>(xpos), static_cast<float>(ypos));
    }
}

void Application::handleMouse(float xpos, float ypos) {
    if (m_FirstMouse) {
        m_LastX = xpos;
        m_LastY = ypos;
        m_FirstMouse = false;
    }

    float xoffset = xpos - m_LastX;
    float yoffset = m_LastY - ypos;

    m_LastX = xpos;
    m_LastY = ypos;

    m_Camera.ProcessMouseMovement(xoffset, yoffset);
}

窗口大小变化处理

如果窗口大小变化也要修改app的大小

否则如果拉大窗口，app不会跟随放大，那么就会出现黑色间隙。

// --- 回调函数 ---
void Application::framebuffer_size_callback(GLFWwindow *window, int width, int height) {
    glViewport(0, 0, width, height);
    Application* app = static_cast<Application*>(glfwGetWindowUserPointer(window));
    if (app) {
        app->m_Width = width;
        app->m_Height = height;
    }
}

渲染

重温变换流程

，(顶点数据)         (gl_Position)                      (最终像素)
   |                  |                                 |
   V                  V                                 V
[模型空间] -> [世界空间] -> [观察空间] -> [裁剪空间] ==> [标准化设备坐标(NDC)] -> [屏幕空间]
    |           |           |           |              |                      |
  Model       View        Projection    Perspective    Viewport
  Matrix      Matrix      Matrix        Division       Transform
 (模型变换)    (视图变换)   (投影变换)     (透视除法)       (视口变换)

<----------------- 在你的顶点着色器中完成 -----------------> <---- GPU 自动完成 ---->

这里准备好投影变换矩阵，和剪裁变换矩阵，交给 world 去渲染

void Application::render() {
    //清理屏幕
    glClearColor(0.1f, 0.1f, 0.1f, 1.0f);
    glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);

    //视图变换，从世界空间到观察空间矩阵
    glm::mat4 view = m_Camera.GetViewMatrix();

    //投影变换，创建相机矩阵，从观察空间到剪裁空间
    glm::mat4 projection = glm::perspective(
            glm::radians(m_Camera.Zoom),//视场角，视野广度
            (float) m_Width / (float) m_Height,//宽高比
            0.1f, 100.0f);//近裁剪平面，远裁剪平面。太近或远的物体不渲染

    // 传给世界去渲染
    m_World.Render(view, projection);
}
