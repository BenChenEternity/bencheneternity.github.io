对象定义

现在定义的主要是对象的逻辑，然后再绑定对象网格。

负责空间状态记录，传递模型矩阵，还有绑定着色器和对象网格。

//=============== 文件路径: objects/Object.h ===============//
#pragma once

#include <vector>
#include <memory>
#include "glm/glm.hpp"
#include "glm/gtc/matrix_transform.hpp"
#include "../components/shader/Shader.h"
#include "../components/mesh/Mesh.h"

// 代表一个可渲染的游戏对象
class Object {
public:
    // 构造函数接收一个共享的Mesh指针
    Object(std::shared_ptr<Mesh> mesh, Shader* shader);
    ~Object() = default;

    // 渲染
    void Render();

    // 设置变换
    void SetPosition(const glm::vec3& position);
    void SetRotation(float angle, const glm::vec3& axis);
    void SetScale(const glm::vec3& scale);

    // 获取模型矩阵和着色器
    glm::mat4 GetModelMatrix() const;
    Shader* GetShader() const;

private:
    std::shared_ptr<Mesh> m_Mesh; // << 持有Mesh的共享指针
    Shader* m_Shader;
    glm::mat4 m_ModelMatrix;

    // 私有变换状态
    glm::vec3 m_Position;
    glm::vec3 m_Scale;
    float m_RotationAngle;
    glm::vec3 m_RotationAxis;

    // 更新模型矩阵
    void updateModelMatrix();
};

实现

//=============== 文件路径: objects/Object.cpp ===============//
#include "Object.h"

Object::Object(std::shared_ptr<Mesh> mesh, Shader *shader)
        : m_Mesh(mesh),
          m_Shader(shader),
          m_ModelMatrix(1.0f),
          m_Position(0.0f),
          m_Scale(1.0f),
          m_RotationAngle(0.0f),
          m_RotationAxis(0.0f, 1.0f, 0.0f) {
}

void Object::Render() {
    // 着色器已由World绑定，通用矩阵已设置
    m_Shader->setMat4("model", m_ModelMatrix);

    m_Mesh->Bind();
    if (m_Mesh->HasIndices()) {
        glDrawElements(GL_TRIANGLES, m_Mesh->GetIndexCount(), GL_UNSIGNED_INT, 0);
    } else {
        glDrawArrays(GL_TRIANGLES, 0, m_Mesh->GetVertexCount());
    }
    m_Mesh->Unbind(); // 实际上，为了性能，可以在批处理结束后再解绑
}

Shader *Object::GetShader() const {
    return m_Shader;
}

void Object::updateModelMatrix() {
    glm::mat4 model = glm::mat4(1.0f);
    model = glm::translate(model, m_Position);
    model = glm::rotate(model, glm::radians(m_RotationAngle), m_RotationAxis);
    model = glm::scale(model, m_Scale);
    m_ModelMatrix = model;
}

void Object::SetPosition(const glm::vec3 &position) {
    m_Position = position;
    updateModelMatrix();
}

void Object::SetRotation(float angle, const glm::vec3 &axis) {
    m_RotationAngle = angle;
    m_RotationAxis = axis;
    updateModelMatrix();
}

void Object::SetScale(const glm::vec3 &scale) {
    m_Scale = scale;
    updateModelMatrix();
}

glm::mat4 Object::GetModelMatrix() const {
    return m_ModelMatrix;
}

对象工厂

创建对象使用工厂模式

//=============== 文件路径: objects/ObjectFactory.h ===============//
#pragma once

#include <string>
#include <memory>
#include <map>
#include <functional>
#include "Object.h"
#include "../components/mesh/MeshManager.h"

class Shader;

// 工厂类，现在管理一个类型注册表
class ObjectFactory {
public:
    // 获取工厂的单例
    static ObjectFactory& GetInstance();

    // 通用创建接口，完全由数据驱动
    std::unique_ptr<Object> createObject(const std::string& type, Shader* shader, MeshManager& meshManager);

    // ---- 注册系统的核心 ----
    using CreatorFunc = std::function<std::unique_ptr<Object>(std::shared_ptr<Mesh>, Shader*)>;

    struct ObjectTypeInfo {
        CreatorFunc creator;
        std::string meshPath;
    };

    // 注册一个新类型
    void registerType(const std::string& type, CreatorFunc creator, const std::string& meshPath);

private:
    ObjectFactory() = default; // 私有构造函数，确保单例
    std::map<std::string, ObjectTypeInfo> m_Registry; // 类型注册表
};


// ---- 自动注册的魔法工具 ----
// 一个模板化的注册器类
template<typename T>
class ObjectRegistrar {
public:
    // 构造函数会把类型信息注册到工厂中
    ObjectRegistrar(const std::string& type, const std::string& meshPath) {
        ObjectFactory::GetInstance().registerType(type,
                                                  [](std::shared_ptr<Mesh> mesh, Shader* shader) -> std::unique_ptr<Object> {
                                                      return std::make_unique<T>(mesh, shader);
                                                  },
                                                  meshPath
        );
    }
};

实现如下

#include "ObjectFactory.h"
#include <iostream>

ObjectFactory& ObjectFactory::GetInstance() {
    static ObjectFactory instance;
    return instance;
}

void ObjectFactory::registerType(const std::string& type, CreatorFunc creator, const std::string& meshPath) {
    m_Registry[type] = { creator, meshPath };
    std::cout << "Info: [ObjectFactory] Registered type '" << type << "'." << std::endl;
}

std::unique_ptr<Object> ObjectFactory::createObject(const std::string& type, Shader* shader, MeshManager& meshManager) {
    auto it = m_Registry.find(type);
    if (it == m_Registry.end()) {
        std::cerr << "Warning: [ObjectFactory] Unknown object type: '" << type << "'. Is it registered?" << std::endl;
        return nullptr;
    }

    const auto& info = it->second;
    std::shared_ptr<Mesh> mesh = meshManager.getOrLoadMesh(type, info.meshPath);
    if (!mesh) {
        std::cerr << "Error: [ObjectFactory] Failed to load mesh for type '" << type << "' from path: " << info.meshPath << std::endl;
        return nullptr;
    }

    return info.creator(mesh, shader);
}

创建具体对象

继承 Object 即可，然后绑定网格

//=============== 文件路径: objects/Cuboid.h ===============//
#pragma once
#include "Object.h"
#include "ObjectFactory.h"
#include "../components/AssetsPath.h"

class Cuboid : public Object {
public:
    Cuboid(std::shared_ptr<Mesh> mesh, Shader* shader);
};

// C++17 魔法：将注册器直接定义在头文件中
// "inline" 关键字告诉链接器，即使多处包含此头文件，也只创建一个 registrar 实例。
// 任何包含了 Cuboid.h 的文件（比如未来的某个关卡逻辑文件），
// 都会创建一个对该注册器的依赖，从而激活链接。
inline static ObjectRegistrar<Cuboid> registrar("Cuboid", AssetPaths::Mesh::CUBE);
