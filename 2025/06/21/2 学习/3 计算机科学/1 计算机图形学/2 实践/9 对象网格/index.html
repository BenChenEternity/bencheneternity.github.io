NicolasDavout 2025-06-21 15:11:53
Categories： Tags：

为什么要对象网格？

为什么不创建对象，然后网格作为成员变量呢？

因为网格是固定的，静态的。如果每个对象拥有一个网格存储一次网格数据，那么非常消耗内存。

如果使用“绑定”的方式，将对象绑定到固定的网格上，也就是所有的对象共用一个网格，极大节省内存。

同时还能分离视图和逻辑，实现解耦。

如果要对象要播放动画呢，岂不是其他同类的对象也要被修改？

  • 对于拉伸、旋转等整体运动：网格是固定的，动画由对象逻辑控制。
  • 对于对象内部的运动：对动画局部设定动作和帧，是多个不同模型（一个模型对应一个网格）在时间上的叠加。

所以并不会印象其他对象

对象网格分三部分：

  • 对象网格类
  • 管理类：管理对象网格
  • 加载类：从文件中加载对象网格

对象网格类定义

现在我们创建的对象网格类 Mesh 是被对象逻辑所绑定的。

定义如下

#pragma once

#include <vector>
#include "GL/glew.h"

class Mesh {
public:
    // 构造函数，接收顶点和可选的索引数据
    Mesh(const std::vector<float>& vertices, const std::vector<unsigned int>& indices = {});
    ~Mesh();

    // 禁用拷贝构造和赋值，因为Mesh管理着GPU资源
    Mesh(const Mesh&) = delete;
    Mesh& operator=(const Mesh&) = delete;

    // 允许移动构造和赋值
    Mesh(Mesh&& other) noexcept;
    Mesh& operator=(Mesh&& other) noexcept;

    // 绑定和解绑VAO
    void Bind() const;
    void Unbind() const;

    // 获取绘制所需的信息
    GLsizei GetIndexCount() const { return m_IndexCount; }
    GLsizei GetVertexCount() const { return m_VertexCount; }
    bool HasIndices() const { return m_HasIndices; }

private:
    void setupMesh(const std::vector<float>& vertices, const std::vector<unsigned int>& indices);
    void clearGPUResources();

    GLuint m_VAO = 0, m_VBO = 0, m_EBO = 0;
    GLsizei m_IndexCount = 0;
    GLsizei m_VertexCount = 0; // 注意：这里的VertexCount是顶点的数量，而不是浮点数的数量
    bool m_HasIndices = false;
};

为什么要禁用拷贝构造和赋值？

因为资源在 GPU 中，如果拷贝是浅拷贝，容易出现错误。

构造中传入的内容

  • 顶点数据，绘制顶点
  • 索引数据，绘制面
#include "Mesh.h"

Mesh::Mesh(const std::vector<float>& vertices, const std::vector<unsigned int>& indices) {
    setupMesh(vertices, indices);
}

Mesh::~Mesh() {
    clearGPUResources();
}

// 移动构造函数
Mesh::Mesh(Mesh&& other) noexcept
        : m_VAO(other.m_VAO), m_VBO(other.m_VBO), m_EBO(other.m_EBO),
          m_IndexCount(other.m_IndexCount), m_VertexCount(other.m_VertexCount), m_HasIndices(other.m_HasIndices) {
    // 将other的资源句柄置零，防止其析构函数释放资源
    other.m_VAO = 0;
    other.m_VBO = 0;
    other.m_EBO = 0;
}

// 移动赋值运算符
Mesh& Mesh::operator=(Mesh&& other) noexcept {
    if (this != &other) {
        // 释放当前对象的资源
        clearGPUResources();

        // 转移资源所有权
        m_VAO = other.m_VAO;
        m_VBO = other.m_VBO;
        m_EBO = other.m_EBO;
        m_IndexCount = other.m_IndexCount;
        m_VertexCount = other.m_VertexCount;
        m_HasIndices = other.m_HasIndices;

        // 将other的资源句柄置零
        other.m_VAO = 0;
        other.m_VBO = 0;
        other.m_EBO = 0;
    }
    return *this;
}


void Mesh::setupMesh(const std::vector<float>& vertices, const std::vector<unsigned int>& indices) {
    m_HasIndices = !indices.empty();
    m_VertexCount = static_cast<GLsizei>(vertices.size() / 6); // 每个顶点6个float (pos+color)
    m_IndexCount = static_cast<GLsizei>(indices.size());

    glGenVertexArrays(1, &m_VAO);
    glGenBuffers(1, &m_VBO);

    glBindVertexArray(m_VAO);

    glBindBuffer(GL_ARRAY_BUFFER, m_VBO);
    glBufferData(GL_ARRAY_BUFFER, vertices.size() * sizeof(float), vertices.data(), GL_STATIC_DRAW);

    if (m_HasIndices) {
        glGenBuffers(1, &m_EBO);
        glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, m_EBO);
        glBufferData(GL_ELEMENT_ARRAY_BUFFER, indices.size() * sizeof(unsigned int), indices.data(), GL_STATIC_DRAW);
    }

    // 位置属性 (location = 0)
    glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 6 * sizeof(float), (void*)0);
    glEnableVertexAttribArray(0);
    // 颜色属性 (location = 1)
    glVertexAttribPointer(1, 3, GL_FLOAT, GL_FALSE, 6 * sizeof(float), (void*)(3 * sizeof(float)));
    glEnableVertexAttribArray(1);

    glBindVertexArray(0);
}

void Mesh::clearGPUResources() {
    // 确保我们只删除有效的句柄
    if (m_VAO != 0) glDeleteVertexArrays(1, &m_VAO);
    if (m_VBO != 0) glDeleteBuffers(1, &m_VBO);
    if (m_EBO != 0) glDeleteBuffers(1, &m_EBO);
}

void Mesh::Bind() const {
    glBindVertexArray(m_VAO);
}

void Mesh::Unbind() const {
    glBindVertexArray(0);
}

网格管理

此处和 shader 的管理方式相同

//=============== 文件路径: components/mesh/MeshManager.h ===============//
#pragma once

#include <string>
#include <map>
#include <memory>
#include "Mesh.h"

class MeshManager {
public:
    MeshManager() = default;

    // 获取一个网格，如果不在缓存中，则尝试从文件加载
    std::shared_ptr<Mesh> getOrLoadMesh(const std::string& name, const std::string& filepath);

private:
    std::map<std::string, std::shared_ptr<Mesh>> m_Meshes;
};
//=============== 文件路径: components/mesh/MeshManager.cpp ===============//
#include "MeshManager.h"
#include "MeshLoader.h"
#include <iostream>

std::shared_ptr<Mesh> MeshManager::getOrLoadMesh(const std::string& name, const std::string& filepath) {
    auto it = m_Meshes.find(name);
    if (it != m_Meshes.end()) {
        return it->second;
    }

    // 如果不在缓存中，调用加载器
    std::cout << "Info: [MeshManager] Cache miss for '" << name << "'. Loading from " << filepath << std::endl;
    auto newMesh = MeshLoader::LoadFromGLB(filepath);
    if (newMesh) {
        m_Meshes[name] = newMesh;
        std::cout << "Info: [MeshManager] Mesh loaded and cached: " << name << std::endl;
    }
    return newMesh;
}

对象网格加载

将 glb 文件加载为对象网格，这里使用到

  • stb_image.h

//=============== 文件路径: components/mesh/MeshLoader.h ===============//
#pragma once

#include <string>
#include <memory>
#include "Mesh.h"

class MeshLoader {
public:
    // 从.glb文件中加载网格
    static std::shared_ptr<Mesh> LoadFromGLB(const std::string& filepath);
};
//=============== 文件路径: components/mesh/MeshLoader.cpp ===============//
#include "MeshLoader.h"
#include <iostream>
#include <vector>

#define TINYGLTF_IMPLEMENTATION
#define STB_IMAGE_IMPLEMENTATION
#define STB_IMAGE_WRITE_IMPLEMENTATION
#include "../../libs/tiny_gltf.h"

std::shared_ptr<Mesh> MeshLoader::LoadFromGLB(const std::string& filepath) {
    tinygltf::Model model;
    tinygltf::TinyGLTF loader;
    std::string err;
    std::string warn;

    bool ret = loader.LoadBinaryFromFile(&model, &err, &warn, filepath.c_str());

    if (!warn.empty()) {
        std::cout << "Warning: [MeshLoader] " << warn << std::endl;
    }
    if (!err.empty()) {
        std::cerr << "Error: [MeshLoader] " << err << std::endl;
        return nullptr;
    }
    if (!ret) {
        std::cerr << "Error: [MeshLoader] Failed to parse glTF" << std::endl;
        return nullptr;
    }

    // --- 我们将从glTF模型中提取顶点和索引数据 ---
    // 这个例子我们简化处理，只加载第一个mesh的第一个primitive
    if (model.meshes.empty()) {
        std::cerr << "Error: [MeshLoader] GLB file has no meshes." << std::endl;
        return nullptr;
    }

    tinygltf::Mesh& gltfMesh = model.meshes[0];
    tinygltf::Primitive& primitive = gltfMesh.primitives[0];

    std::vector<float> vertices;
    std::vector<unsigned int> indices;

    // 1. 加载索引 (Indices)
    const tinygltf::Accessor& indexAccessor = model.accessors[primitive.indices];
    const tinygltf::BufferView& indexBufferView = model.bufferViews[indexAccessor.bufferView];
    const tinygltf::Buffer& indexBuffer = model.buffers[indexBufferView.buffer];

    const unsigned char* indexBufferStart = indexBuffer.data.data() + indexBufferView.byteOffset + indexAccessor.byteOffset;
    size_t indexCount = indexAccessor.count;

    indices.reserve(indexCount);
    // 根据索引类型 (unsigned short, int, etc.) 读取
    if (indexAccessor.componentType == TINYGLTF_COMPONENT_TYPE_UNSIGNED_SHORT) {
        const auto* p = reinterpret_cast<const unsigned short*>(indexBufferStart);
        for(size_t i = 0; i < indexCount; ++i) indices.push_back(p[i]);
    } else if (indexAccessor.componentType == TINYGLTF_COMPONENT_TYPE_UNSIGNED_INT) {
        const auto* p = reinterpret_cast<const unsigned int*>(indexBufferStart);
        for(size_t i = 0; i < indexCount; ++i) indices.push_back(p[i]);
    }
    //... 可以添加对其他类型的支持

    // 2. 加载顶点属性 (Vertex Attributes)
    const float* positionBuffer = nullptr;
    const float* normalBuffer = nullptr;
    const float* texcoordBuffer = nullptr;
    size_t vertexCount = 0;

    if (primitive.attributes.count("POSITION")) {
        const tinygltf::Accessor& accessor = model.accessors[primitive.attributes["POSITION"]];
        const tinygltf::BufferView& view = model.bufferViews[accessor.bufferView];
        positionBuffer = reinterpret_cast<const float*>(&(model.buffers[view.buffer].data[view.byteOffset + accessor.byteOffset]));
        vertexCount = accessor.count;
    }
    // ... 可以用类似方法获取法线 (NORMAL) 和纹理坐标 (TEXCOORD_0)

    // 我们将把所有顶点数据打包到一个`vertices`数组中
    // 我们的新顶点格式: Pos(3) + Color(3) = 6 floats
    // (为了保持简单，我们暂时不加载法线和UV，颜色用白色硬编码)
    vertices.reserve(vertexCount * 6);
    for (size_t i = 0; i < vertexCount; ++i) {
        // Position
        vertices.push_back(positionBuffer[i * 3 + 0]);
        vertices.push_back(positionBuffer[i * 3 + 1]);
        vertices.push_back(positionBuffer[i * 3 + 2]);
        // Color (Hardcoded to white)
        vertices.push_back(1.0f);
        vertices.push_back(1.0f);
        vertices.push_back(1.0f);
    }

    return std::make_shared<Mesh>(vertices, indices);
}
